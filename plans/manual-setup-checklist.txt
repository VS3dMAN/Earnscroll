STEP 3 — Make SENTRY_AUTH_TOKEN available at build time
---------------------------------------------------------
.env has the token but Gradle won't auto-read it. Without this,
builds succeed but Sentry stack traces will be unreadable (minified).

  Option A (local builds): export the variable in your shell before building
    Windows CMD:  set SENTRY_AUTH_TOKEN=<your_token>
    PowerShell:   $env:SENTRY_AUTH_TOKEN="<your_token>"
    bash/Git Bash: export SENTRY_AUTH_TOKEN=<your_token>

  Option B (EAS builds): add as an EAS secret
    Run: eas secret:create --scope project --name SENTRY_AUTH_TOKEN --value <your_token>

  Your token is already in e:\Builds\EarnScroll\.env — copy it from there.

-- Table 1: analytics_consent
CREATE TABLE analytics_consent (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL UNIQUE REFERENCES auth.users(id) ON DELETE CASCADE,
  diagnostics_enabled BOOLEAN NOT NULL DEFAULT false,
  analytics_enabled BOOLEAN NOT NULL DEFAULT false,
  consent_given_at TIMESTAMPTZ,
  consent_revoked_at TIMESTAMPTZ,
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);
ALTER TABLE analytics_consent ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can manage own consent" ON analytics_consent
  FOR ALL USING (auth.uid() = user_id) WITH CHECK (auth.uid() = user_id);

-- Table 2: analytics_events
CREATE TABLE analytics_events (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  event_name TEXT NOT NULL,
  event_properties JSONB DEFAULT '{}',
  screen_name TEXT,
  session_id UUID,
  app_version TEXT,
  platform TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);
ALTER TABLE analytics_events ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can insert own events" ON analytics_events
  FOR INSERT WITH CHECK (auth.uid() = user_id);
CREATE POLICY "Users can view own events" ON analytics_events
  FOR SELECT USING (auth.uid() = user_id);

-- Table 3: diagnostic_logs
CREATE TABLE diagnostic_logs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  level TEXT NOT NULL CHECK (level IN ('info', 'warn', 'error')),
  message TEXT NOT NULL,
  context JSONB DEFAULT '{}',
  app_version TEXT,
  platform TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()

ALTER TABLE diagnostic_logs ENABLE ROW LEVEL SECURITY;

  FOR INSERT WITH CHECK (auth.uid() = user_id);
CREATE POLICY "Users can view own logs" ON diagnostic_logs
  FOR SELECT USING (auth.uid() = user_id);

STEP 6 — Decide on withAppBlocker plugin
------------------------------------------
Two accessibility-service plugins exist but only one is registered in app.json.

  Situation:
  - plugins/withEarnScrollNative.js  → REGISTERED in app.json (injects BlockerService)
  - plugins/withAppBlocker.js        → NOT registered (would inject AppBlockerService)
  - Both do similar things — having both would register two accessibility services

  Decision options:
    Option A (recommended): Keep only withEarnScrollNative (already registered).
      → Delete plugins/withAppBlocker.js and native-src/ folder if unused.

    Option B: Switch to withAppBlocker.
      → Add "./plugins/withAppBlocker" to app.json plugins array, remove withEarnScrollNative,
        re-run prebuild.

    Option C: Keep both (not recommended — two accessibility services for same purpose).

  After deciding and making any changes, re-run: npx expo prebuild --clean
